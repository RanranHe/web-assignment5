function binary(target, arr, l, r) {
    if (l <= r) { //when l index>= r index stop searching
        if (arr[l] == target) //if(value of l index is target just return this index)
            return l;
        if (arr[r] == target) // same way return the right index
            return r;
        var mid = Math.ceil((r + l) / 2); // get int number not double
        if (arr[mid] == target) {
            return mid; //get it
        } else if (arr[mid] > target) {
            return binary(target, arr, l, mid - 1); //if tareget<= mid ,just search in left part of array
        } else {
            return binary(target, arr, mid + 1, r); //just search in right part.
        }
    }
    return -1; // can not find target in array return -1
}
